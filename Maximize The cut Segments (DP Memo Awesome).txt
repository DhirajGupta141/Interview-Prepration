Given an integer N denoting the Length of a line segment. You need to cut the line segment in such a way that the cut length of a line segment each time is either x , y or z. Here x, y, and z are integers.
After performing all the cut operations, your total number of cut segments must be maximum.

Example 1:

Input:
N = 4
x = 2, y = 1, z = 1
Output: 4
Explanation:Total length is 4, and the cut
lengths are 2, 1 and 1.  We can make
maximum 4 segments each of length 1.
Example 2:

Input:
N = 5
x = 5, y = 3, z = 2
Output: 2
Explanation: Here total length is 5, and
the cut lengths are 5, 3 and 2. We can
make two segments of lengths 3 and 2.
___________________________________________________________SOLUTION__________________________________

class Solution
{
    
    int dfs(int n,int x,int y,int z,int t[])
    {
        if(n==0)
            return 0;
        
        if(t[n]!=-1)
            return t[n];
            
        int answerFromX=Integer.MIN_VALUE;
        int answerFromY=Integer.MIN_VALUE;
        int answerFromZ=Integer.MIN_VALUE;
        
        //trying all possibilities
        if(n>=x)
            answerFromX=dfs(n-x,x,y,z,t);
        
        if(n>=y)
            answerFromY=dfs(n-y,x,y,z,t);
        
        if(n>=z)
            answerFromZ=dfs(n-z,x,y,z,t);
         
        //storing so that we could return the value if same call happens again   
        t[n]=1+Math.max(answerFromX,Math.max(answerFromY,answerFromZ)); //+1 for current cut
        
        return t[n];
            
    }
    public int maximizeCuts(int n, int x, int y, int z)
    {
        int t[]=new int[100005];
        Arrays.fill(t,-1);
        int ans=dfs(n,x,y,z,t);
        if(ans<0)
            return 0; 
        return ans;
    }
}
