Given a linked list and a value x, partition it such that all nodes less than x come before nodes greater than or equal to x.

You should preserve the original relative order of the nodes in each of the two partitions.

For example,
Given 1->4->3->2->5->2 and x = 3,
return 1->2->2->4->3->5.

____________________________________Solution___________________________
public class Solution 
{
    
    public ListNode partition(ListNode head, int x)
    {
        if(head==null || head.next==null)
        {
            return head;
        }
        
        ListNode small_head=new ListNode(-1);
        ListNode small=small_head;
        ListNode large_head=new ListNode(-1);
        ListNode large=large_head;
        
        while(head!=null)
        {
            if(head.val<x)
            {
                small.next=new ListNode(head.val);
                small=small.next;
            }
            else
            {
                large.next=new ListNode(head.val);
                large=large.next;
            }
            
            head=head.next;
        }
        
        large.next=null;
        
        small.next=large_head.next;
        
        return small_head.next;
        
    }
}
