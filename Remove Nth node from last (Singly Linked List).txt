Given a linked list, remove the nth node from the end of list and return its head.

For example,
Given linked list: 1->2->3->4->5, and n = 2.
After removing the second node from the end, the linked list becomes 1->2->3->5.

 Note:
If n is greater than the size of the list, remove the first node of the list.
Try doing it using constant additional space.

___________________________________________________________________SOLUTION USING TWO PASS_____________________________________________
public class Solution
{
    public ListNode removeNthFromEnd(ListNode head, int n) 
    {
        if(head==null || n==0)
        {
            return head;
        }
        
        ListNode current=head;   //current pointing node;
        
        int count=0;    //for length of linked list
        while(current!=null)
        {
            count++;
            current=current.next;
        }
        
        if(n>=count)    //checking for edge case 
        {
            head=head.next;
            return head;
        }
        
        int t=count-n;    //again the time for which we need to traverse
        
        ListNode previous=null;  // previous to node which we need to delete
        
        current=head;           
        
        for(int i=0;i<t;i++)   //reaching to the node which we need to delete
        {
            previous=current;
            current=current.next;
        }
        
        previous.next=current.next;        //making the forward link by skipping the node
        
        return head;   //return ans :)
        
    }
}
