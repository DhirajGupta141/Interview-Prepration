class Solution
{
    
    static void dfs(int node,ArrayList<ArrayList<Integer>> adj,int visited[], Stack<Integer>ans)
    {
        visited[node]=1;
        //here we go --> we are not adding the node to the answer we are going depth of node
        for(int n:adj.get(node))
        {   
            if(visited[n]==0)
                dfs(n,adj,visited,ans);
        }
//after the calls get over for the node that is dfs() (depth) over we add that node in ans

        ans.add(node);        
    }
    
    
    //Function to return list containing vertices in Topological order. 
    static int[] topoSort(int V, ArrayList<ArrayList<Integer>> adj) 
    {
        Stack<Integer> ans = new Stack<>();
        
        int visited[]=new int[V];
        //for all the component
        for(int i=0;i<V;i++)
        {
            if(visited[i]==0)
                dfs(i,adj,visited,ans);//making call only if it is note visited
        }
        
        int sorted_ans[]=new int[V]; //our final answer
        
        int pointer=0;
        
        while(ans.size()>0)
            sorted_ans[pointer++]=ans.pop();
        
        return sorted_ans;
    }
}
