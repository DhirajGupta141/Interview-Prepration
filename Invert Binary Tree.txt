public class Solution 
{
    void invert(TreeNode root)
    {
        if(root==null)
        {
            return;
        }
        
        invert(root.left);
        invert(root.right);
        
        TreeNode temp=root.left;
        root.left=root.right;
        root.right=temp;
        
    }
    public TreeNode invertTree(TreeNode root)
    {
        if(root==null)
        {
            return null;
        }
        
        invert(root);
        return root;
        
    }
}
