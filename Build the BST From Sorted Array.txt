Given an array where elements are sorted in ascending order, convert it to a height balanced BST.

 Balanced tree : a height-balanced binary tree is defined as a binary tree in which the depth of the two subtrees of every node never differ by more than 1. 
Example :


Given A : [1, 2, 3]
A height balanced BST  : 

      2
    /   \
   1     3

__________________________________Just Build The Tree__________________
public class Solution 
{
    //Building the tree
    TreeNode helper(List<Integer> a,int low,int high)
    {
        if(low>high)
        {
            return null;
        }
        
        int  mid=(low+high)/2;
        
        TreeNode root=new TreeNode(a.get(mid));  //creating the root node
        root.left=helper(a,low,mid-1);  //left subtree
        root.right=helper(a,mid+1,high);  //right subtree
       
        return root;
        
    }
    
    public TreeNode sortedArrayToBST(final List<Integer> a)
    {
        if(a==null || a.size()==0)
        {
            return null;
        }
        
        if(a.size()==1)
        {
            return new TreeNode(a.get(0));
        }
        int n=a.size();
        
        TreeNode root=helper(a,0,n-1);   // a low high
        
        return root;
    }
}
