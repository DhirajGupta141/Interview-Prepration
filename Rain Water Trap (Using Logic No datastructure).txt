Problem Description

Given an integer array A of non-negative integers representing an elevation map where the width of each bar is 1, 
compute how much water it is able to trap after raining.



Problem Constraints
1 <= |A| <= 100000



Input Format
The only argument given is integer array A.



Output Format
Return the total water it is able to trap after raining.



Example Input
Input 1:

 A = [0, 1, 0, 2, 1, 0, 1, 3, 2, 1, 2, 1]
Input 2:

 A = [1, 2]


Example Output
Output 1:

 6
Output 2:

 0


Example Explanation
Explanation 1:

 
 In this case, 6 units of rain water (blue section) are being trapped.
Explanation 2:

 No water is trapped.
______________________________________SOLUTION_______________________________
public class Solution
{
    // DO NOT MODIFY THE ARGUMENTS WITH "final" PREFIX. IT IS READ ONLY
    public int trap(final int[] height) 
    {
        int n = height.length; 
        int left=0; 
        int right=n-1;
        int res=0;
        int maxleft=0, maxright=0; // maximum height building to left and right of
                                    //current index building
        
        while(left<=right) //representing the array index inshort the building number
        {
            
            if(height[left]<=height[right]) //comparing the end nodes
                                             //deciding where water can go?
            {
                
                if(height[left]>=maxleft)  // if current building height is greater than
                                        // or equals to the overall max till now
                { 
                    maxleft=height[left];  //updating to max_till_now to curent building
                }
                else   //else we could store the water
                {
                    res+=maxleft-height[left];
                }
                
                left++;
            }
            
            else  //left side bulding is larger
            {
                
                if(height[right]>=maxright)
                {
                    maxright= height[right];
                }
                else
                { 
                    res+=maxright-height[right];
                }
                
                right--;
            }
        }
        return res;
    }
}
