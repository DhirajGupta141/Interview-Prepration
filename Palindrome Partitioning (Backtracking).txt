Given a string s, partition s such that every substring of the partition is a palindrome. Return all possible palindrome partitioning of s.

A palindrome string is a string that reads the same backward as forward.

 

Example 1:

Input: s = "aab"
Output: [["a","a","b"],["aa","b"]]
Example 2:

Input: s = "a"
Output: [["a"]]

Input:
S = "geeks"
Output:
g e e k s
g ee k s
Explanation:
All possible palindromic partitions
are printed.
Example 2:

Input:
S = "madam"
Output:
m a d a m
m ada m
madam

______________________________________________________SOLUTION_______________________________________________

TC: O(N*2^N)   //each N has choice either it could be a part of the palindrome or not so 2^N and N for the palindorme check so overall its O(N*2^N)
SC: O(N^2)
class Solution
{
    void dfs(int idx,String s , List<List<String>> ans, List<String> subans)
    {
        
        //base case increasing recursion
        if(idx==s.length()) //tried all posibiities
        {
            ans.add(new ArrayList<>(subans));
            return;
        }
        
        for(int i=idx;i<s.length();i++)  //doing partitian at each possible index
        {
            if(isPal(s,idx,i)) //checking if the partition made is palindrome or not
            {
                subans.add(s.substring(idx,i+1));  //adding to subans if it is a palindrome  --> current substring want to be a part that is selected 
                
                dfs(i+1,s,ans,subans);
                
                subans.remove(subans.size()-1);  //backtrack  <-- removing the last added character   --> current substring dont want to be a part that is not selected 
            }
        }
        
    }
    
    
    boolean isPal(String s,int start,int end)
    {
        while(start<=end)
        {
            if(s.charAt(start)==s.charAt(end))
            {
                start++;
                end--;
            }
            else
            {
                return false;
            }
        }
        
        return true;
    }
    
    
    
    public List<List<String>> partition(String s)
    {
        
        List<List<String>> ans=new ArrayList<>();
        
        List<String>subans=new ArrayList<>();
        dfs(0,s,ans,subans);
        
        return ans;
    }
}
